```js

function _changeData(data = []) {
    const res = [];
    const map = {};

    data.forEach((item) => {
    map[item.id] = item;
    if (item.parent in map) { 
        const target = map[item.parent];
        target.children || (target.children = []);

        target.children.push(item);
    } else {
        res.push(item);
    }
    });

    return res;
}

console.log(
    this._changeData([
        { id: 1 },
        { id: 2, parent: 1 },
        { id: 3, parent: 2 },
        { id: 4, parent: 1 },
        { id: 5, parent: 2 },
        { id: 6, parent: 1 },
        { id: 7, parent: 6 },
    ])
);
```
